```typescript
private async findByRequestId(requestId: string): Promise<boolean> {
  const response = await this.sheets.spreadsheets.values.get({
    auth: this.auth,
    spreadsheetId: this.spreadsheetId,
    range: `${this.sheetNames.transactions}!F:F`,
  });
  const ids = (response.data.values || []).flat();
  return ids.includes(requestId);
}
```

**2. Category validation with logging (CAT-05):**

Add import at the top:
```typescript
import { EXPENSE_CATEGORIES } from '@/app/services/llm/constants';
```

After `const parsedData = await llmService.parseTransaction(transaction);`, add a validation pass:
```typescript
const validatedData = parsedData.map(tx => {
  if (!EXPENSE_CATEGORIES.includes(tx.category as typeof EXPENSE_CATEGORIES[number])) {
    console.warn(`[API] Unknown category "${tx.category}" for transaction "${tx.motive}". Falling back to Uncategorized.`);
    return { ...tx, category: 'Uncategorized' };
  }
  return tx;
});
```

Use `validatedData` (not `parsedData`) in the `appendTransaction` call and in the response summary.

**3. Uncategorized fallback on InvalidInputError (CAT-06):**

Replace the existing `InvalidInputError` catch block (currently returns 400) with a fallback that persists to Sheets:
```typescript
if (error instanceof InvalidInputError) {
  console.warn(`[API] Failed to parse transaction: "${transaction.slice(0, 100)}". Logging as Uncategorized.`);
  const fallbackTx = {
    motive: transaction.slice(0, 500),
    amount: 0,
    type: 'Expense' as const,
    category: 'Uncategorized',
  };
  try {
    await spreadsheetService.appendTransaction([fallbackTx], requestId);
    return NextResponse.json(
      { message: 'Could not parse expense details. Logged as Uncategorized for manual review.' },
      { status: 200 }
    );
  } catch (sheetError) {
    console.error('[API] Failed to log Uncategorized fallback:', sheetError);
    return NextResponse.json(
      { message: 'Failed to log expense' },
      { status: 500 }
    );
  }
}
```

The `requestId` variable from the top of the handler is in scope here. The fallback's `requestId` is the same UUID, making even failed parses idempotent on retry.

**Complete catch order in the inner try/catch must be:**
1. `DuplicateRequestError` → 409
2. `InvalidInputError` → Uncategorized fallback → 200 (or 500 on sheet failure)
3. `SpreadsheetWriteError` → 500
4. re-throw unexpected errors
